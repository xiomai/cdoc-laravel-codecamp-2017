Blog Model and Migration


> php artisan list 
# View list of available artisan commands 

> php artisan make:model -h 
# View options for the command

> # php artisan make:model --migration --controller Blog 
> php artisan make:model --mc Blog 

# with that, Laravel created for us a Model, the Controller of that model
# and the migration which we should modify first before runing the Migration
Model is singular to the table name

Column Types


Schema::create('blogs', function (Blueprint $table) {
    $table->increments('id');
    $table->integer('user_id');
    $table->string('title');
    $table->text('content');
    $table->timestamps();
});
